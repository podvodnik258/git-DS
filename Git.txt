1. Настройка
git --version 	проверка версии

git config --global user.name
git config --global user.email	глобальный пользователь
git config user.name
git config user.email	локальный пользовате ль

2. Инициализация

git init	 инициализация папки для работы внутри
git add . 	[файл, папка]	добавить всю папку[файл, папка] в стадию ожидания commit
git add css/*.html	все расширением html
git add css/**/*.html	все расширением html во вложенных папках
git add !css/style.css	кроме этого файла
git status	смотреть статус
git rm --cached file	убрать из стадии ожидания
git commit -m "comment"	Сохранить в локальный репозиторий
git log 		просмотр изменений
git log --oneline	просмотр изменений в одну строку
.gitignore		файл в корне для игнорирования изменений файлов /folder/

3. Старые версии

git log --oneline	запоминаем id коммита
git checkout e7ce969	просмотр к старому коммиту (желательно для только просмотра)
git checkout master	возврат к последней версии
git revert e7ce969	отменить выбранный коммит (отменяются действия)
git reset e7ce969	удаляет коммиты после этого, но остаемся в этой версии
git reset --hard e7ce969	возврат к этой версии удаляя после нее 

4. Ветки

git branch forum	создает ветку forum
git checkout forum	перейти в ветку forum
git branch -a 		показать все ветки
git checkout master	переход в основную ветку
git checkout -b admin	создаем и переходим в ветку admin
git checkout master 
git merge forum		слияние ветки forum к master	


5. Введение в GitHub. Работа с удаленным репозиторием

git remote add origin https://github.com/podvodnik258/git-DS.git	подключаем удаленный репозиторий
git push -u origin master					записать в удаленный репозиторий ветку мастер	
git clone https://github.com/podvodnik258/git-DS.git		клонировать все с гит хаба
git pull					    		Второй сотрудник актуализация нового.
